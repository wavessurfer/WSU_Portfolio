Hands-On Project 13-3-Adding Input Devices to a Circuit

Do Hands-On Project 13-3
Below is the code that is needed for step 5. 

// C++ code
//
int dacValue=0;
void setup()
{
  Serial.begin(9600);
  pinMode(4,OUTPUT);
  pinMode(13, OUTPUT);
  pinMode(A1, INPUT);
}

void loop()
{
  dacValue= analogRead(A1);
  Serial.println(dacValue);
  if(dacValue >= 650) {
    tone(4,2000);
    digitalWrite(13, HIGH);
  } else  {
    digitalWrite(13, LOW);
    noTone(4);
  }
     
}
